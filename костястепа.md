## üß© 1. –ö–ª–∞—Å—Å `SOM` ‚Äî —Å–ª–æ–π –ö–æ—Ö–æ–Ω–µ–Ω–∞ (Self-Organizing Map)

```python
class SOM:
    def __init__(self, input_dim, grid_rows=5, grid_cols=5, lr=0.5, sigma=None, n_iter=100):
        self.input_dim = input_dim              # —Ä–∞–∑–º–µ—Ä–Ω–æ—Å—Ç—å –≤—Ö–æ–¥–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö
        self.grid_rows = grid_rows              # –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å—Ç—Ä–æ–∫ –Ω–µ–π—Ä–æ–Ω–Ω–æ–π —Å–µ—Ç–∫–∏
        self.grid_cols = grid_cols              # –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å—Ç–æ–ª–±—Ü–æ–≤ –Ω–µ–π—Ä–æ–Ω–Ω–æ–π —Å–µ—Ç–∫–∏
        self.num_neurons = grid_rows * grid_cols  # –æ–±—â–µ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –Ω–µ–π—Ä–æ–Ω–æ–≤
        self.lr = lr                            # —Å–∫–æ—Ä–æ—Å—Ç—å –æ–±—É—á–µ–Ω–∏—è
        self.n_iter = n_iter                    # –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∏—Ç–µ—Ä–∞—Ü–∏–π
        self.sigma = sigma if sigma else max(grid_rows, grid_cols) / 4  # —Ä–∞–¥–∏—É—Å —Å–æ—Å–µ–¥—Å—Ç–≤–∞
        self.weights = X[np.random.choice(len(X), self.num_neurons, replace=False)]  
        # –∏–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –≤–µ—Å–æ–≤ —Å–ª—É—á–∞–π–Ω—ã–º–∏ –æ–±—Ä–∞–∑—Ü–∞–º–∏ –∏–∑ –¥–∞–Ω–Ω—ã—Ö
```

–ö–∞–∂–¥—ã–π –Ω–µ–π—Ä–æ–Ω –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è–µ—Ç ‚Äú—Ç–æ—á–∫—É‚Äù –≤ –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ –ø—Ä–∏–∑–Ω–∞–∫–æ–≤.

```python
    def _find_bmu(self, x):
        dists = np.linalg.norm(self.weights - x, axis=1)  # –≤—ã—á–∏—Å–ª—è–µ–º –µ–≤–∫–ª–∏–¥–æ–≤—ã —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è –æ—Ç x –¥–æ –≤—Å–µ—Ö –≤–µ—Å–æ–≤
        return np.argmin(dists)  # –≤–æ–∑–≤—Ä–∞—â–∞–µ–º –∏–Ω–¥–µ–∫—Å –Ω–µ–π—Ä–æ–Ω–∞-–ø–æ–±–µ–¥–∏—Ç–µ–ª—è (BMU ‚Äî Best Matching Unit)
```

üéØ **–ß—Ç–æ –¥–µ–ª–∞–µ—Ç:**
–ù–∞—Ö–æ–¥–∏—Ç –Ω–µ–π—Ä–æ–Ω, —á–µ–π –≤–µ—Å –Ω–∞–∏–±–æ–ª–µ–µ –ø–æ—Ö–æ–∂ –Ω–∞ –≤—Ö–æ–¥–Ω–æ–π –≤–µ–∫—Ç–æ—Ä.
–≠—Ç–æ –∫–ª—é—á–µ–≤–∞—è –æ–ø–µ—Ä–∞—Ü–∏—è SOM ‚Äî –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ –ø–æ–±–µ–¥–∏—Ç–µ–ª—è.

```python
    def _neighborhood(self, bmu_idx, iter_no):
        sigma = self.sigma * np.exp(-iter_no / self.n_iter)  # —Ä–∞–¥–∏—É—Å —Å–æ—Å–µ–¥—Å—Ç–≤–∞ —É–º–µ–Ω—å—à–∞–µ—Ç—Å—è —Å–æ –≤—Ä–µ–º–µ–Ω–µ–º
        bmu_r = bmu_idx // self.grid_cols  # —Å—Ç—Ä–æ–∫–∞ –ø–æ–±–µ–¥–∏—Ç–µ–ª—è
        bmu_c = bmu_idx % self.grid_cols   # —Å—Ç–æ–ª–±–µ—Ü –ø–æ–±–µ–¥–∏—Ç–µ–ª—è
        neighborhood = np.zeros(self.num_neurons)
        for idx in range(self.num_neurons):
            r = idx // self.grid_cols
            c = idx % self.grid_cols
            dist_sq = (bmu_r - r)**2 + (bmu_c - c)**2  # –∫–≤–∞–¥—Ä–∞—Ç —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è –º–µ–∂–¥—É –Ω–µ–π—Ä–æ–Ω–∞–º–∏ –Ω–∞ —Å–µ—Ç–∫–µ
            neighborhood[idx] = np.exp(-dist_sq / (2*sigma**2))  # –≤–µ—Å —Å–æ—Å–µ–¥—Å—Ç–≤–∞ (—á–µ–º –±–ª–∏–∂–µ ‚Äî —Ç–µ–º –±–æ–ª—å—à–µ –≤–ª–∏—è–Ω–∏–µ)
        return neighborhood
```

üåê **–ß—Ç–æ –¥–µ–ª–∞–µ—Ç:**
–†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ—Ç, –Ω–∞—Å–∫–æ–ª—å–∫–æ —Å–∏–ª—å–Ω–æ –±—É–¥–µ—Ç –æ–±–Ω–æ–≤–ª—è—Ç—å—Å—è –∫–∞–∂–¥—ã–π –Ω–µ–π—Ä–æ–Ω, –≤ –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–∏ –æ—Ç —Ç–æ–≥–æ, –∫–∞–∫ –¥–∞–ª–µ–∫–æ –æ–Ω –Ω–∞—Ö–æ–¥–∏—Ç—Å—è –æ—Ç –ø–æ–±–µ–¥–∏—Ç–µ–ª—è –Ω–∞ —Å–µ—Ç–∫–µ.
–≠—Ç–æ —Å–æ–∑–¥–∞—ë—Ç ‚Äú–º—è–≥–∫–æ–µ‚Äù –≤–ª–∏—è–Ω–∏–µ ‚Äî –Ω–µ —Ç–æ–ª—å–∫–æ –ø–æ–±–µ–¥–∏—Ç–µ–ª—å, –Ω–æ –∏ –µ–≥–æ —Å–æ—Å–µ–¥–∏ –Ω–µ–º–Ω–æ–≥–æ –¥–≤–∏–≥–∞—é—Ç—Å—è –∫ –≤—Ö–æ–¥—É.

```python
    def fit(self, X, noise_std=0.01):
        for iter_no in range(self.n_iter):                 # –æ—Å–Ω–æ–≤–Ω–æ–π —Ü–∏–∫–ª –æ–±—É—á–µ–Ω–∏—è
            idx = np.random.permutation(len(X))            # –ø–µ—Ä–µ–º–µ—à–∏–≤–∞–µ–º –¥–∞–Ω–Ω—ã–µ –¥–ª—è —Å—Ç–æ—Ö–∞—Å—Ç–∏—á–µ—Å–∫–æ–≥–æ –æ–±—É—á–µ–Ω–∏—è
            for i in idx:
                x = X[i] + np.random.normal(0, noise_std, size=X[i].shape)  # –¥–æ–±–∞–≤–ª—è–µ–º –Ω–µ–±–æ–ª—å—à–æ–π —à—É–º –¥–ª—è —É—Å—Ç–æ–π—á–∏–≤–æ—Å—Ç–∏
                bmu_idx = self._find_bmu(x)                # –∏—â–µ–º –ø–æ–±–µ–¥–∏—Ç–µ–ª—è
                h = self._neighborhood(bmu_idx, iter_no)   # –≤—ã—á–∏—Å–ª—è–µ–º –≤–ª–∏—è–Ω–∏–µ —Å–æ—Å–µ–¥–µ–π
                self.weights += self.lr * h[:, np.newaxis] * (x - self.weights)
                # –æ–±–Ω–æ–≤–ª—è–µ–º –≤–µ—Å–∞: –¥–≤–∏–≥–∞–µ–º –∏—Ö –±–ª–∏–∂–µ –∫ –≤—Ö–æ–¥—É —Å –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç–æ–º –≤–ª–∏—è–Ω–∏—è h
```

## üß† 2. –ö–ª–∞—Å—Å `GrossbergLayer` ‚Äî —Å–ª–æ–π –ì—Ä–æ—Å—Å–±–µ—Ä–≥–∞ (–∞—Å—Å–æ—Ü–∏–∞—Ç–∏–≤–Ω—ã–π —Å–ª–æ–π)

```python
class GrossbergLayer:
    def __init__(self, num_neurons, num_classes, lr=0.01, n_iter=50):
        self.weights = np.random.rand(num_neurons, num_classes)  # –≤–µ—Å–∞ –æ—Ç –Ω–µ–π—Ä–æ–Ω–æ–≤ SOM –∫ –∫–ª–∞—Å—Å–∞–º
        self.lr = lr
        self.n_iter = n_iter
```

üß© **–ß—Ç–æ –¥–µ–ª–∞–µ—Ç:**
–ö–∞–∂–¥–æ–º—É –Ω–µ–π—Ä–æ–Ω—É SOM —Å–æ–ø–æ—Å—Ç–∞–≤–ª—è–µ—Ç –≤–µ–∫—Ç–æ—Ä –≤–µ—Å–æ–≤, —É–∫–∞–∑—ã–≤–∞—é—â–∏–π, –Ω–∞—Å–∫–æ–ª—å–∫–æ –æ–Ω —Å–≤—è–∑–∞–Ω —Å –∫–∞–∂–¥—ã–º –∫–ª–∞—Å—Å–æ–º.
–≠—Ç–æ —Å–ª–æ–π ‚Äú–∞—Å—Å–æ—Ü–∏–∞—Ü–∏–π‚Äù: –Ω–µ–π—Ä–æ–Ω ‚Üí –∫–ª–∞—Å—Å.

---

```python
    def fit(self, bmu_indices, labels):
        Y = np.eye(self.weights.shape[1])[labels]  # one-hot –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏–µ –º–µ—Ç–æ–∫ –∫–ª–∞—Å—Å–æ–≤
        for _ in range(self.n_iter):
            idx = np.random.permutation(len(bmu_indices))  # –ø–µ—Ä–µ–º–µ—à–∏–≤–∞–µ–º –¥–∞–Ω–Ω—ã–µ
            for i in idx:
                bmu = bmu_indices[i]                       # –∏–Ω–¥–µ–∫—Å –∞–∫—Ç–∏–≤–Ω–æ–≥–æ –Ω–µ–π—Ä–æ–Ω–∞ (–∏–∑ SOM)
                y_true = Y[i]                              # –ø—Ä–∞–≤–∏–ª—å–Ω—ã–π –∫–ª–∞—Å—Å
                self.weights[bmu] += self.lr * (y_true - self.weights[bmu])
                # –æ–±–Ω–æ–≤–ª—è–µ–º —Ç–æ–ª—å–∫–æ –≤–µ—Å–∞ –ø–æ–±–µ–¥–∏—Ç–µ–ª—è, –ø—Ä–∏–±–ª–∏–∂–∞—è –µ–≥–æ –∫ –ø—Ä–∞–≤–∏–ª—å–Ω–æ–º—É –∫–ª–∞—Å—Å—É
```

–ï—Å–ª–∏ –Ω–µ–π—Ä–æ–Ω-–ø–æ–±–µ–¥–∏—Ç–µ–ª—å –∞–∫—Ç–∏–≤–∏—Ä–æ–≤–∞–ª—Å—è –ø—Ä–∏ –ø—Ä–∏–º–µ—Ä–µ –∫–ª–∞—Å—Å–∞ ‚ÄúIris-setosa‚Äù, –µ–≥–æ –≤–µ—Å–∞ –∫–æ—Ä—Ä–µ–∫—Ç–∏—Ä—É—é—Ç—Å—è –≤ —Å—Ç–æ—Ä–æ–Ω—É one-hot –≤–µ–∫—Ç–æ—Ä–∞ —ç—Ç–æ–≥–æ –∫–ª–∞—Å—Å–∞.
–ü–æ—Å—Ç–µ–ø–µ–Ω–Ω–æ –Ω–µ–π—Ä–æ–Ω ‚Äú–∑–∞–∫—Ä–µ–ø–ª—è–µ—Ç—Å—è‚Äù –∑–∞ –∫–æ–Ω–∫—Ä–µ—Ç–Ω—ã–º –∫–ª–∞—Å—Å–æ–º.

## ‚öôÔ∏è 3. –ö–ª–∞—Å—Å `CPN` ‚Äî –æ–±—ä–µ–¥–∏–Ω—ë–Ω–Ω–∞—è —Å–µ—Ç—å (Counterpropagation Network)

```python
class CPN:
    def __init__(self, input_dim, kohonen_grid=(5,5), grossberg_out=3, lr_som=0.5, n_iter_som=100, lr_gross=0.1, n_iter_gross=50):
        self.kohonen = SOM(input_dim, grid_rows=kohonen_grid[0], grid_cols=kohonen_grid[1],
                           lr=lr_som, n_iter=n_iter_som)
        self.grossberg = GrossbergLayer(self.kohonen.num_neurons, grossberg_out,
                                        lr=lr_gross, n_iter=n_iter_gross)
        self.train_acc_history = []
        self.test_acc_history = []
```

üîó **–ß—Ç–æ –¥–µ–ª–∞–µ—Ç:**
–°–æ–∑–¥–∞—ë—Ç –¥–≤–∞ —É—Ä–æ–≤–Ω—è:

1. **Kohonen (SOM)** ‚Äî –æ–±—É—á–∞–µ—Ç—Å—è –±–µ–∑ —É—á–∏—Ç–µ–ª—è (–∫–ª–∞—Å—Ç–µ—Ä–∏–∑–∞—Ü–∏—è –¥–∞–Ω–Ω—ã—Ö).
2. **Grossberg** ‚Äî –æ–±—É—á–∞–µ—Ç—Å—è —Å —É—á–∏—Ç–µ–ª–µ–º (–∞—Å—Å–æ—Ü–∏–∞—Ü–∏—è –∫–ª–∞—Å—Ç–µ—Ä–æ–≤ —Å –º–µ—Ç–∫–∞–º–∏ –∫–ª–∞—Å—Å–æ–≤).

```python
    def fit(self, X_train, y_train, X_test=None, y_test=None):
        self.kohonen.fit(X_train)                     # —Å–Ω–∞—á–∞–ª–∞ —Å–∞–º–æ–æ—Ä–≥–∞–Ω–∏–∑–∞—Ü–∏—è (–±–µ–∑ –º–µ—Ç–æ–∫)
        bmu_indices_train = self.kohonen.predict(X_train)  # —Å–æ—Ö—Ä–∞–Ω—è–µ–º, –∫–∞–∫–æ–π –Ω–µ–π—Ä–æ–Ω –ø–æ–±–µ–¥–∏–ª –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –æ–±—Ä–∞–∑—Ü–∞

        for epoch in range(self.grossberg.n_iter):    # —Ç–µ–ø–µ—Ä—å –æ–±—É—á–µ–Ω–∏–µ —Å —É—á–∏—Ç–µ–ª–µ–º (Grossberg)
            self.grossberg.fit(bmu_indices_train, y_train)

            y_train_pred = self.predict(X_train)      # –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏—è –Ω–∞ –æ–±—É—á–∞—é—â–µ–º –Ω–∞–±–æ—Ä–µ
            self.train_acc_history.append(accuracy_score(y_train, y_train_pred))

            if X_test is not None and y_test is not None:  # –µ—Å–ª–∏ –µ—Å—Ç—å —Ç–µ—Å—Ç–æ–≤—ã–µ –¥–∞–Ω–Ω—ã–µ
                y_test_pred = self.predict(X_test)
                self.test_acc_history.append(accuracy_score(y_test, y_test_pred))
```

üöÄ **–ö–∞–∫ –ø—Ä–æ—Ö–æ–¥–∏—Ç –æ–±—É—á–µ–Ω–∏–µ:**

1. **SOM:** —Ñ–æ—Ä–º–∏—Ä—É–µ—Ç –∫–∞—Ä—Ç—É –ø—Ä–∏–∑–Ω–∞–∫–æ–≤ –ø–æ –≤—Ö–æ–¥–Ω—ã–º –¥–∞–Ω–Ω—ã–º.
2. **Grossberg:** —Å–≤—è–∑—ã–≤–∞–µ—Ç –∫–∞–∂–¥—ã–π –Ω–µ–π—Ä–æ–Ω –∫–∞—Ä—Ç—ã —Å –∫–æ–Ω–∫—Ä–µ—Ç–Ω—ã–º –∫–ª–∞—Å—Å–æ–º.
3. **Accuracy:** —Å—á–∏—Ç–∞–µ—Ç —Ç–æ—á–Ω–æ—Å—Ç—å –ø–æ—Å–ª–µ –∫–∞–∂–¥–æ–π —ç–ø–æ—Ö–∏ –æ–±—É—á–µ–Ω–∏—è, —á—Ç–æ–±—ã –æ—Ç—Å–ª–µ–∂–∏–≤–∞—Ç—å –ø—Ä–æ–≥—Ä–µ—Å—Å.
